// ===== JUEGO DE DEBUGGING CON COMENTARIOS =====
// Este código debería mostrar un contador que cuenta de 1 a 5.
// Si encuentras un error, córregelo editando mentalmente el código.

function iniciarContador() {
    let contador = 0;
    let mensaje = "";
    
    /* 
     * El bucle debería ejecutarse 5 veces, sumando 1 en cada iteración.
     * PISTA 1: Revisa la condición del bucle for.
     */
    for (let i = 0; i < 5; i++) {
        contador++; // Incrementamos el contador
        mensaje += "Contador: " + contador + "<br>";
    }
    
    // PISTA 2: ¿Estamos mostrando el mensaje correctamente?
    // TODO: El elemento donde mostrar el resultado tiene id="resultado"
    document.getElementById("resultados").innerHTML = mensaje;
    
    /* PISTA 3: La función document.getElementById() busca un elemento
     * con el id exacto que le pasamos.
     * Si el id no existe, devolverá null y causará un error.
     */
}
Pista adicional: El error está en el ID utilizado para obtener el elemento donde mostrar el resultado. En el código se usa "resultados" pero el elemento correcto tiene el ID "resultado" (sin la 's' final).
